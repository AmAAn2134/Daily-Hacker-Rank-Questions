#include <stdio.h>

int find_unique_digit_product(int n) {
    int m=n;
            int p=1;
    if(n==0){
        return 0;
    }
    for(int i=0;i<=9;i++){
        n=m;
        while(n!=0){
            int rem =n % 10;
            if(i==rem){
                p=p*i;
                break;
            }
            n/=10;
        }
    }
    return p;
}

int main() {
    int n;
    scanf("%d", &n);
    printf("%d\n", find_unique_digit_product(n));
    return 0;
}
